You do not have to put every interaction inside the header itself. Because the table exposes engine state, a nearby toolbar can reorder columns, resize them, and reflect the current layout.

## External layout controls

```tsx live wide file=App.tsx
import { useState } from 'react'

import { Button } from '@/components/ui/button'
import { DataTable, DataTableCell, DataTableColumn, DataTableColumnHeader } from '@/components/ui/data-table'
import { columns$, useEngineRef, useRemoteCellValue, useRemotePublisher } from '@virtuoso.dev/data-table'
import { reorderColumns$ } from '@virtuoso.dev/data-table/column-reorder'
import { resizeColumn$ } from '@virtuoso.dev/data-table/column-resize'

import { products } from './data'

import type { ColumnInfo } from '@virtuoso.dev/data-table'

function findColumnKey(columns: Map<string, ColumnInfo> | undefined, field: string) {
  return columns ? [...columns].find(([, info]) => info.field === field)?.[0] : undefined
}

export default function App() {
  const engineRef = useEngineRef()
  const columns = useRemoteCellValue(columns$, engineRef)
  const reorderColumns = useRemotePublisher(reorderColumns$, engineRef)
  const resizeColumn = useRemotePublisher(resizeColumn$, engineRef)
  const [draftWidth, setDraftWidth] = useState('280')

  const order = columns ? Array.from(columns.values(), (info) => info.field).join(' -> ') : 'loading'
  const nameKey = findColumnKey(columns, 'name')
  const regionKey = findColumnKey(columns, 'region')
  const statusKey = findColumnKey(columns, 'status')

  return (
    <div className="space-y-4">
      <div className="rounded-lg border bg-muted/40 p-3 text-sm text-muted-foreground">{order}</div>

      <div className="flex flex-wrap items-center gap-2">
        <Button
          variant="outline"
          disabled={!statusKey || !nameKey}
          onClick={() => {
            if (statusKey && nameKey) {
              reorderColumns({ sourceKey: statusKey, targetKey: nameKey, position: 'before' })
            }
          }}
        >
          Move status first
        </Button>
        <Button
          variant="outline"
          disabled={!regionKey || !statusKey}
          onClick={() => {
            if (regionKey && statusKey) {
              reorderColumns({ sourceKey: regionKey, targetKey: statusKey, position: 'after' })
            }
          }}
        >
          Move region after status
        </Button>
        <input
          className="h-9 w-24 rounded-md border bg-background px-3 text-sm"
          onChange={(event) => setDraftWidth(event.target.value)}
          type="number"
          value={draftWidth}
        />
        <Button
          variant="outline"
          disabled={!nameKey}
          onClick={() => {
            const width = Number(draftWidth)
            if (nameKey && Number.isFinite(width)) {
              resizeColumn({ key: nameKey, width })
            }
          }}
        >
          Resize product column
        </Button>
      </div>

      <DataTable className="rounded-xl" data={{ data: products, groups: [] }} engineRef={engineRef} style={{ height: 400 }}>
        <DataTableColumn field="name">
          <DataTableColumnHeader>Product</DataTableColumnHeader>
          <DataTableCell className="font-medium">{({ row }) => row.data.name}</DataTableCell>
        </DataTableColumn>
        <DataTableColumn field="category">
          <DataTableColumnHeader>Category</DataTableColumnHeader>
          <DataTableCell>{({ cellValue }) => String(cellValue)}</DataTableCell>
        </DataTableColumn>
        <DataTableColumn field="status">
          <DataTableColumnHeader>Status</DataTableColumnHeader>
          <DataTableCell>{({ cellValue }) => String(cellValue)}</DataTableCell>
        </DataTableColumn>
        <DataTableColumn field="region">
          <DataTableColumnHeader>Region</DataTableColumnHeader>
          <DataTableCell>{({ cellValue }) => String(cellValue)}</DataTableCell>
        </DataTableColumn>
      </DataTable>
    </div>
  )
}
```

export const products = Array.from({ length: 100 }, (_, index) => {
  const stock = (index * 5) % 32

  return {
    id: `SKU-${String(index + 1).padStart(3, '0')}`,
    name: `Product ${index + 1}`,
    category: ['Office', 'Peripherals', 'Audio'][index % 3]!,
    status: stock < 6 ? 'Backorder' : stock < 12 ? 'Low stock' : 'In stock',
    region: ['US', 'EU', 'APAC'][index % 3]!,
  }
})

## Notes

- Reorder actions work in terms of internal column keys, so read those keys from `columns$` rather than guessing.
- `resizeColumn$` comes from the opt-in `@virtuoso.dev/data-table/column-resize` sub-entry, so tables that never resize headers do not pay for that reducer.
- Width overrides are useful when business users want a denser or roomier view without rewriting the column renderer.
