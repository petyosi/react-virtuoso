Column reordering is an opt-in feature module. Import the model streams from
`@virtuoso.dev/data-table/column-reorder` and use the shadcn registry components for drag UI.

```tsx live wide
import { Button } from '@/components/ui/button'
import { DataTable, DataTableCell, DataTableColumn, DataTableColumnHeader } from '@/components/ui/data-table'
import { localModel, columns$, useEngineRef, useRemoteCellValue, useRemotePublisher } from '@virtuoso.dev/data-table'
import { reorderColumns$ } from '@virtuoso.dev/data-table/column-reorder'

import type { ColumnInfo } from '@virtuoso.dev/data-table'

const rows = Array.from({ length: 40 }, (_, index) => ({
  id: `SKU-${String(index + 1).padStart(3, '0')}`,
  name: `Product ${index + 1}`,
  status: ['In stock', 'Low stock', 'Backorder'][index % 3]!,
  region: ['US', 'EU', 'APAC'][index % 3]!,
}))

function findColumnKey(columns: Map<string, ColumnInfo> | undefined, field: string) {
  return columns ? [...columns].find(([, column]) => column.field === field)?.[0] : undefined
}

const model = localModel({ data: rows })

export default function App() {
  const engineRef = useEngineRef()
  const columns = useRemoteCellValue(columns$, engineRef)
  const reorderColumns = useRemotePublisher(reorderColumns$, engineRef)
  const statusKey = findColumnKey(columns, 'status')
  const nameKey = findColumnKey(columns, 'name')

  return (
    <div className="space-y-3">
      <Button
        disabled={!statusKey || !nameKey}
        onClick={() => {
          if (statusKey && nameKey) {
            reorderColumns({ sourceKey: statusKey, targetKey: nameKey, position: 'before' })
          }
        }}
        variant="outline"
      >
        Move status before product
      </Button>
      <DataTable className="rounded-xl" computeRowKey={({ data }) => data.id} model={model} engineRef={engineRef} style={{ height: 320 }}>
        <DataTableColumn field="name">
          <DataTableColumnHeader>Product</DataTableColumnHeader>
          <DataTableCell className="font-medium">{({ cellValue }) => String(cellValue)}</DataTableCell>
        </DataTableColumn>
        <DataTableColumn field="status">
          <DataTableColumnHeader>Status</DataTableColumnHeader>
          <DataTableCell>{({ cellValue }) => String(cellValue)}</DataTableCell>
        </DataTableColumn>
        <DataTableColumn field="region">
          <DataTableColumnHeader>Region</DataTableColumnHeader>
          <DataTableCell>{({ cellValue }) => String(cellValue)}</DataTableCell>
        </DataTableColumn>
      </DataTable>
    </div>
  )
}
```

The underlying stream works from any control surface:

```tsx
import { columns$, useEngineRef, useRemoteCellValue, useRemotePublisher } from '@virtuoso.dev/data-table'
import { reorderColumns$ } from '@virtuoso.dev/data-table/column-reorder'

const engineRef = useEngineRef()
const columns = useRemoteCellValue(columns$, engineRef)
const reorderColumns = useRemotePublisher(reorderColumns$, engineRef)

const entries = [...(columns ?? new Map())]
const statusKey = entries.find(([, column]) => column.field === 'status')?.[0]
const nameKey = entries.find(([, column]) => column.field === 'name')?.[0]

if (statusKey && nameKey) {
  reorderColumns({ sourceKey: statusKey, targetKey: nameKey, position: 'before' })
}
```

Use `reorderColumnGroup$` when a header interaction moves a contiguous group of columns. The
`DraggableGroupHeader` registry component wires that behavior for `ColumnGroupHeader`.

Related reorder state:

- `reorderColumns$` moves one column
- `reorderColumnGroup$` moves a group of columns while preserving internal order
- `columnDragState$`, `beginColumnDrag$`, `endColumnDrag$`, and `setColumnDropTarget$` coordinate drag UI
- `columnOrderPersistenceAdapter()` stores order by field name
