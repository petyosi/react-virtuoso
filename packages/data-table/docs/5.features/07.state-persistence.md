State persistence is opt-in. Mount `DataTableStatePersistence` inside the table and pass the feature
adapters you want to save.

```tsx live wide
import * as React from 'react'

import { Button } from '@/components/ui/button'
import {
  DataTable,
  DataTableCell,
  DataTableColumn,
  DataTableColumnHeader,
  DataTableGroupHeader,
  HeaderEdge,
  HeaderOverlay,
  HeaderStart,
} from '@/components/ui/data-table'
import { ReorderDropZone, ReorderGrip } from '@/components/ui/data-table/column-reorder'
import { ResizeHandle } from '@/components/ui/data-table/column-resize'
import { columns$, remoteModel, useCellValue, usePublisher } from '@virtuoso.dev/data-table'
import { columnOrderPersistenceAdapter } from '@virtuoso.dev/data-table/column-reorder'
import { columnWidthPersistenceAdapter } from '@virtuoso.dev/data-table/column-resize'
import {
  columnVisibilityPersistenceAdapter,
  columnVisibilityState$,
  setColumnVisibility$,
} from '@virtuoso.dev/data-table/column-visibility'
import { DataTableStatePersistence, modelStatePersistenceAdapter } from '@virtuoso.dev/data-table/state-persistence'

import type { DataResult, GroupHeaderRenderParams, MessageEnvelope, ModelPersistenceState } from '@virtuoso.dev/data-table'

type DatasetKey = 'people' | 'orders' | 'metrics'
type SortField = 'id' | 'name' | 'status' | ''
type StatusFilter = 'Active' | 'Pending' | 'Paid' | 'Open' | ''
type GroupField = 'status' | 'city' | ''
type RemoteDataRow = Record<string, number | string>
type RemoteRow = RemoteDataRow | { groupLabel: string }

const datasets: Record<DatasetKey, { label: string; rows: RemoteDataRow[] }> = {
  people: {
    label: 'People',
    rows: Array.from({ length: 36 }, (_, index) => ({
      id: `USR-${String(index + 1).padStart(3, '0')}`,
      name: `User ${index + 1}`,
      status: ['Active', 'Pending', 'Paused'][index % 3]!,
      city: ['Sofia', 'Berlin', 'Madrid', 'Oslo'][index % 4]!,
    })),
  },
  orders: {
    label: 'Orders',
    rows: Array.from({ length: 30 }, (_, index) => ({
      id: `ORD-${String(index + 1).padStart(3, '0')}`,
      customer: `Customer ${index + 1}`,
      status: ['Paid', 'Open', 'Overdue'][index % 3]!,
      total: `$${(160 + index * 19).toLocaleString()}`,
    })),
  },
  metrics: {
    label: 'Metrics',
    rows: Array.from({ length: 28 }, (_, index) => ({
      id: `MET-${String(index + 1).padStart(3, '0')}`,
      name: `Metric ${index + 1}`,
      score: 80 + (index % 17),
      trend: ['Up', 'Flat', 'Down'][index % 3]!,
    })),
  },
}

function wait(ms: number) {
  return new Promise((resolve) => window.setTimeout(resolve, ms))
}

function isGroupRow(row: RemoteRow): row is { groupLabel: string } {
  return 'groupLabel' in row
}

function fieldsFromResult(result: DataResult<RemoteRow>) {
  const seen = new Set<string>()
  const fields: string[] = []

  for (const row of result.data) {
    if (isGroupRow(row)) {
      continue
    }
    for (const field of Object.keys(row)) {
      if (!seen.has(field)) {
        seen.add(field)
        fields.push(field)
      }
    }
  }

  return fields
}

function compareRows(field: string) {
  return (left: RemoteDataRow, right: RemoteDataRow) => String(left[field] ?? '').localeCompare(String(right[field] ?? ''))
}

function groupRows(rows: RemoteDataRow[], field: GroupField) {
  if (!field) {
    return { rows, groups: [] }
  }

  const buckets = new Map<string, RemoteDataRow[]>()
  for (const row of rows) {
    const key = String(row[field] ?? '(blank)')
    buckets.set(key, [...(buckets.get(key) ?? []), row])
  }

  const groupedRows: RemoteRow[] = []
  const groups: { index: number; level: number }[] = []
  for (const key of [...buckets.keys()].sort((left, right) => left.localeCompare(right))) {
    const bucketRows = buckets.get(key)!
    groups.push({ index: groupedRows.length, level: 0 })
    groupedRows.push({ groupLabel: `${field}: ${key} (${bucketRows.length})` })
    groupedRows.push(...bucketRows)
  }

  return { rows: groupedRows, groups }
}

async function fetchRows({ limit, offset, params, signal }) {
  await wait(160)
  if (signal.aborted) {
    throw new Error('aborted')
  }

  let rows = [...datasets[params.dataset].rows]
  if (params.filterStatus) {
    rows = rows.filter((row) => row.status === params.filterStatus)
  }
  if (params.sortBy) {
    rows.sort(compareRows(params.sortBy))
  }

  const result = groupRows(rows, params.groupBy)
  return {
    groups: result.groups,
    rows: result.rows.slice(offset, offset + limit),
    totalCount: result.rows.length,
  }
}

function createModel() {
  return remoteModel<RemoteRow, { dataset: DatasetKey; filterStatus?: StatusFilter; groupBy?: GroupField; sortBy?: SortField }>({
    fetch: fetchRows,
    initialParams: { dataset: 'people' },
    pageSize: 50,
    actions: {
      dataset: {
        handler: ({ payload, params }) => ({ ...params, dataset: payload as DatasetKey }),
      },
      filterStatus: {
        handler: ({ payload, params }) => ({ ...params, filterStatus: payload as StatusFilter }),
        persistence: { isEmpty: (value) => value === '', key: 'status' },
      },
      groupBy: {
        handler: ({ payload, params }) => ({ ...params, groupBy: payload as GroupField }),
        persistence: { isEmpty: (value) => value === '' },
      },
      sort: {
        handler: ({ payload, params }) => ({ ...params, sortBy: payload as SortField }),
        persistence: { isEmpty: (value) => value === '' },
      },
    },
  })
}

function currentModelState(model): ModelPersistenceState {
  return model.persistence?.capture('default', null) ?? { version: 1, actions: {} }
}

function useModelState(model) {
  const [state, setState] = React.useState<ModelPersistenceState>(() => currentModelState(model))

  React.useEffect(() => {
    setState(currentModelState(model))
    return model.persistence?.subscribe('default', () => setState(currentModelState(model)))
  }, [model])

  return state
}

function useDiscoveredFields(model) {
  const [fields, setFields] = React.useState<string[]>([])

  React.useEffect(() => {
    return model.subscribe((message: MessageEnvelope) => {
      if (message.type === 'result') {
        setFields(fieldsFromResult(message.payload as DataResult<RemoteRow>))
      }
    })
  }, [model])

  return fields
}

function actionValue<T extends string>(state: ModelPersistenceState, key: string, fallback: T): T {
  const value = state.actions[key]
  return typeof value === 'string' ? (value as T) : fallback
}

function QueryControls({ dataset, model, onDatasetChange, onReset }) {
  const state = useModelState(model)
  const sort = actionValue<SortField>(state, 'sort', '')
  const status = actionValue<StatusFilter>(state, 'status', '')
  const groupBy = actionValue<GroupField>(state, 'groupBy', '')

  return (
    <div className="space-y-3">
      <div className="flex flex-wrap gap-2">
        {(Object.keys(datasets) as DatasetKey[]).map((key) => (
          <Button key={key} onClick={() => onDatasetChange(key)} size="sm" variant={dataset === key ? 'default' : 'outline'}>
            {datasets[key].label}
          </Button>
        ))}
        <Button onClick={onReset} size="sm" variant="outline">
          Reset saved state
        </Button>
      </div>

      <div className="flex flex-wrap gap-2">
        {(['', 'id', 'name', 'status'] as SortField[]).map((field) => (
          <Button
            key={field || 'none'}
            onClick={() => model.send({ action: 'sort', payload: field })}
            size="sm"
            variant={sort === field ? 'default' : 'outline'}
          >
            {field ? `Sort ${field}` : 'No sort'}
          </Button>
        ))}
      </div>

      <div className="flex flex-wrap gap-2">
        {(['', 'Active', 'Pending', 'Paid', 'Open'] as StatusFilter[]).map((value) => (
          <Button
            key={value || 'all'}
            onClick={() => model.send({ action: 'filterStatus', payload: value })}
            size="sm"
            variant={status === value ? 'default' : 'outline'}
          >
            {value ? `Status ${value}` : 'All statuses'}
          </Button>
        ))}
      </div>

      <div className="flex flex-wrap gap-2">
        {(['', 'status', 'city'] as GroupField[]).map((field) => (
          <Button
            key={field || 'none'}
            onClick={() => model.send({ action: 'groupBy', payload: field })}
            size="sm"
            variant={groupBy === field ? 'default' : 'outline'}
          >
            {field ? `Group ${field}` : 'No grouping'}
          </Button>
        ))}
      </div>

      <div className="rounded-md border bg-muted/30 p-3 text-xs text-muted-foreground">
        <span className="font-medium text-foreground">Persisted model actions:</span>{' '}
        {Object.keys(state.actions).length === 0 ? 'none' : JSON.stringify(state.actions)}
      </div>
    </div>
  )
}

function ColumnVisibilityPicker() {
  const columns = useCellValue(columns$)
  const visibility = useCellValue(columnVisibilityState$)
  const setColumnVisibility = usePublisher(setColumnVisibility$)

  return (
    <div className="flex flex-wrap gap-2 rounded-md border bg-muted/30 p-3">
      {[...columns].map(([key, column]) => {
        const visible = visibility.get(key) ?? column.visible !== false
        return (
          <Button
            key={key}
            aria-pressed={visible}
            onClick={() => setColumnVisibility({ key, visible: !visible })}
            size="sm"
            variant={visible ? 'default' : 'outline'}
          >
            {column.field}
          </Button>
        )
      })}
    </div>
  )
}

function RemoteColumns({ fields }: { fields: string[] }) {
  return (
    <>
      {fields.map((field) => (
        <DataTableColumn key={field} field={field}>
          <DataTableColumnHeader className="min-w-32">
            <HeaderStart component={ReorderGrip} />
            <HeaderOverlay component={ReorderDropZone} />
            <HeaderEdge component={ResizeHandle} />
            {() => field}
          </DataTableColumnHeader>
          <DataTableCell>{({ cellValue }) => String(cellValue)}</DataTableCell>
        </DataTableColumn>
      ))}
    </>
  )
}

function RemoteGroupHeader() {
  return (
    <DataTableGroupHeader>
      {({ row }: GroupHeaderRenderParams) => (isGroupRow(row.data as RemoteRow) ? (row.data as { groupLabel: string }).groupLabel : '')}
    </DataTableGroupHeader>
  )
}

const adapters = [
  columnVisibilityPersistenceAdapter(),
  columnOrderPersistenceAdapter(),
  columnWidthPersistenceAdapter(),
  modelStatePersistenceAdapter(),
]

export default function App() {
  const [resetVersion, setResetVersion] = React.useState(0)
  const [dataset, setDataset] = React.useState<DatasetKey>('people')
  const model = React.useMemo(createModel, [])
  const fields = useDiscoveredFields(model)

  const switchDataset = React.useCallback(
    (nextDataset: DatasetKey) => {
      setDataset(nextDataset)
      model.send({ action: 'dataset', payload: nextDataset })
    },
    [model]
  )

  return (
    <div className="space-y-4">
      <QueryControls
        dataset={dataset}
        model={model}
        onDatasetChange={switchDataset}
        onReset={() => setResetVersion((version) => version + 1)}
      />

      <div className="rounded-md border bg-muted/30 p-3 text-xs text-muted-foreground">
        <span className="font-medium text-foreground">Discovered fields:</span> {fields.length === 0 ? 'loading' : fields.join(', ')}
      </div>

      <DataTable className="rounded-xl" model={model} style={{ height: 420 }}>
        <DataTableStatePersistence adapters={adapters} resetKey={resetVersion} storageKey="docs-combined-state-persistence-feature-page" />
        <ColumnVisibilityPicker />
        <RemoteGroupHeader />
        <RemoteColumns fields={fields} />
      </DataTable>
    </div>
  )
}
```

```tsx
import { DataTableStatePersistence, modelStatePersistenceAdapter } from '@virtuoso.dev/data-table/state-persistence'
import { columnOrderPersistenceAdapter } from '@virtuoso.dev/data-table/column-reorder'
import { columnWidthPersistenceAdapter } from '@virtuoso.dev/data-table/column-resize'
import { columnVisibilityPersistenceAdapter } from '@virtuoso.dev/data-table/column-visibility'

const adapters = [
  columnVisibilityPersistenceAdapter(),
  columnOrderPersistenceAdapter(),
  columnWidthPersistenceAdapter(),
  modelStatePersistenceAdapter(),
]

<DataTable model={model} style={{ height: 420 }}>
  <DataTableStatePersistence adapters={adapters} storageKey="inventory-table" />
  {/* columns */}
</DataTable>
```

The default storage is `window.localStorage`. Pass `storage` when you need `sessionStorage`, a test
double, or an application-specific persistence layer.

Keep the `adapters` array stable. Define it outside the component or wrap it in `useMemo()` so the
persistence component does not resubscribe on every render.

## Available adapters

- `modelStatePersistenceAdapter()` persists action state exposed by `localModel()` and
  `remoteModel()`.
- `columnOrderPersistenceAdapter()` persists column order by field name.
- `columnWidthPersistenceAdapter()` persists resize overrides by field name.
- `columnVisibilityPersistenceAdapter()` persists visibility overrides by field name.

Use `resetKey` to clear stored state when the user resets the view or when your app changes the
meaning of the saved table.

```tsx
<DataTableStatePersistence adapters={adapters} resetKey={resetVersion} storageKey="inventory-table" />
```

Adapters preserve unmatched fields where possible, so state can survive partially matching dynamic
schemas. For generated columns, stable `field` names are the persistence contract.

Model persistence only stores actions that opt in with `persistence`. For local models, add it to a
pipeline action. For remote models, add it to the action that transforms query params.

```tsx
const model = remoteModel({
  // ...
  actions: {
    filter: {
      persistence: true,
      handler: ({ params, payload }) => ({ ...params, status: payload }),
    },
  },
})
```

See the [State Persistence example](/data-table/examples/state-persistence/) for a remote table with
model and column-width persistence wired together.
