This example shows a two-level hierarchy: region first, then category inside each region.

## APIs that matter

- `GroupHeaderCell` for group rows
- `groups` markers inside `TableData`

```tsx live wide file=App.tsx
import { DataTable, DataTableCell, DataTableColumn, DataTableColumnHeader, GroupHeaderCell } from '@/components/ui/data-table'

import { groupedProducts } from './data'

export default function App() {
  return (
    <DataTable className="rounded-xl" data={groupedProducts} style={{ height: 420 }}>
      <GroupHeaderCell>
        {({ level, row }) => (
          <div className={level === 0 ? 'bg-muted px-3 py-2 font-semibold' : 'bg-muted/50 px-6 py-2 font-medium'}>
            {(row.data as { label: string }).label}
          </div>
        )}
      </GroupHeaderCell>

      <DataTableColumn field="name">
        <DataTableColumnHeader>Product</DataTableColumnHeader>
        <DataTableCell className="font-medium">{({ row }) => row.data.name}</DataTableCell>
      </DataTableColumn>
      <DataTableColumn field="status">
        <DataTableColumnHeader>Status</DataTableColumnHeader>
        <DataTableCell>{({ cellValue }) => String(cellValue)}</DataTableCell>
      </DataTableColumn>
      <DataTableColumn field="region">
        <DataTableColumnHeader>Region</DataTableColumnHeader>
        <DataTableCell>{({ cellValue }) => String(cellValue)}</DataTableCell>
      </DataTableColumn>
    </DataTable>
  )
}
```

```ts live file=data.ts
interface Product {
  id: string
  name: string
  category: 'Office' | 'Peripherals' | 'Audio'
  region: 'US' | 'EU'
  status: 'In stock' | 'Low stock' | 'Backorder'
}

interface GroupLabel {
  label: string
}

const categories: Product['category'][] = ['Office', 'Peripherals', 'Audio']
const regions: Product['region'][] = ['US', 'EU']

const products: Product[] = Array.from({ length: 90 }, (_, index) => {
  const stock = (index * 5) % 32

  return {
    id: `SKU-${String(index + 1).padStart(3, '0')}`,
    name: `${categories[index % categories.length]} Item ${index + 1}`,
    category: categories[index % categories.length]!,
    region: regions[index % regions.length]!,
    status: stock < 6 ? 'Backorder' : stock < 12 ? 'Low stock' : 'In stock',
  }
})

export const groupedProducts: {
  data: (GroupLabel | Product)[]
  groups: { index: number; level: number }[]
} = (() => {
  const data: (GroupLabel | Product)[] = []
  const groups: { index: number; level: number }[] = []

  for (const region of regions) {
    groups.push({ index: data.length, level: 0 })
    data.push({ label: `Region: ${region}` })

    for (const category of categories) {
      const rows = products.filter((product) => product.region === region && product.category === category)
      groups.push({ index: data.length, level: 1 })
      data.push({ label: `${category} (${rows.length})` })
      data.push(...rows)
    }
  }

  return { data, groups }
})()
```

## Tradeoffs

Deeply nested groups are expressive, but they also increase cognitive load. Keep the group labels concise and obvious.
