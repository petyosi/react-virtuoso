The table exposes the same kind of imperative control you often need in admin dashboards:

- jump to a specific row after a search result
- keep the current scroll location visible in nearby UI
- scroll an item into view without rebuilding the whole dataset

## Controlled scrolling

```tsx live wide file=App.tsx
import { useRef, useState } from 'react'

import { Button } from '@/components/ui/button'
import { DataTable, DataTableCell, DataTableColumn, DataTableColumnHeader } from '@/components/ui/data-table'

import { products } from './data'

import type { ListScrollLocation, VirtuosoDataTableMethods } from '@virtuoso.dev/data-table'

const currency = new Intl.NumberFormat('en-US', {
  currency: 'USD',
  style: 'currency',
})

export default function App() {
  const tableRef = useRef<VirtuosoDataTableMethods<(typeof products)[number]>>(null)
  const [scrollLocation, setScrollLocation] = useState<ListScrollLocation | null>(null)

  return (
    <div className="space-y-4">
      <div className="flex flex-wrap gap-2">
        <Button onClick={() => tableRef.current?.scrollToRow({ index: 0, behavior: 'smooth' })} variant="outline">
          Top
        </Button>
        <Button onClick={() => tableRef.current?.scrollToRow({ index: 48, align: 'center', behavior: 'smooth' })} variant="outline">
          Center row 49
        </Button>
        <Button onClick={() => tableRef.current?.scrollIntoView({ index: 96, align: 'end', behavior: 'smooth' })} variant="outline">
          Bring row 97 into view
        </Button>
      </div>

      <div className="rounded-lg border bg-muted/40 px-3 py-2 text-sm text-muted-foreground">
        {scrollLocation === null
          ? 'Scroll state will appear here.'
          : `visible height: ${Math.round(scrollLocation.visibleListHeight)}px, bottom offset: ${Math.round(scrollLocation.bottomOffset)}px`}
      </div>

      <DataTable
        ref={tableRef}
        className="rounded-xl"
        data={{ data: products, groups: [] }}
        initialLocation={{ index: 24, align: 'start-no-overflow' }}
        onScroll={setScrollLocation}
        style={{ height: 400 }}
      >
        <DataTableColumn field="name">
          <DataTableColumnHeader>Product</DataTableColumnHeader>
          <DataTableCell className="font-medium">{({ row }) => row.data.name}</DataTableCell>
        </DataTableColumn>
        <DataTableColumn field="category">
          <DataTableColumnHeader>Category</DataTableColumnHeader>
          <DataTableCell>{({ cellValue }) => String(cellValue)}</DataTableCell>
        </DataTableColumn>
        <DataTableColumn field="price">
          <DataTableColumnHeader className="justify-end">Price</DataTableColumnHeader>
          <DataTableCell className="text-right tabular-nums">{({ cellValue }) => currency.format(Number(cellValue))}</DataTableCell>
        </DataTableColumn>
      </DataTable>
    </div>
  )
}
```

```ts live file=data.ts
export const products = Array.from({ length: 120 }, (_, index) => ({
  id: `SKU-${String(index + 1).padStart(3, '0')}`,
  name: `Product ${index + 1}`,
  category: ['Office', 'Peripherals', 'Audio', 'Software'][index % 4]!,
  price: 49 + (index % 11) * 13,
}))
```

## Notes

- `initialLocation` is the easiest way to start at a useful place without rendering everything above it.
- `scrollToRow` is best when you already know the exact target row.
- `scrollIntoView` is better for preserving context because it only moves when necessary.
