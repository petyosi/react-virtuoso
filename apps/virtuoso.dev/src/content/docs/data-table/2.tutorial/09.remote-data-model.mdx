`remoteSource()` pushes the expensive work out of the view layer. The table tells the model which rows the viewport needs, and the model fetches only the required slice.

Offset mode is a good fit for classic filtering and sorting over a known total count. Append mode is better for cursor-style feeds or "load more" experiences.

## Offset and append in one demo

```tsx live wide file=App.tsx
import { useMemo, useState } from 'react'

import { DataTable, DataTableCell, DataTableColumn, DataTableColumnHeader } from '@/components/ui/data-table'
import { defaultAppendViewportHandler, defaultOffsetViewportHandler, remoteSource } from '@virtuoso.dev/data-table'

import { appendPlaceholder, fetchAppendProducts, fetchOffsetProducts, offsetPlaceholder } from './api'

import type { Product, Query } from './api'

const currency = new Intl.NumberFormat('en-US', {
  currency: 'USD',
  style: 'currency',
})

export default function App() {
  const [mode, setMode] = useState<'offset' | 'append'>('offset')
  const [category, setCategory] = useState('all')

  const model = useMemo(() => {
    if (mode === 'offset') {
      return remoteSource<Product, Query>({
        actions: {
          filter: {
            handler: ({ params, payload }) => ({ ...params, category: payload as string }),
          },
        },
        fetch: fetchOffsetProducts,
        initialParams: { category: 'all' },
        onViewportChange: defaultOffsetViewportHandler,
        pageSize: 40,
        placeholder: offsetPlaceholder,
      })
    }

    return remoteSource<Product, Query>({
      actions: {
        filter: {
          handler: ({ params, payload }) => ({ ...params, category: payload as string }),
        },
      },
      fetch: fetchAppendProducts,
      initialParams: { category: 'all' },
      mode: 'append',
      onViewportChange: defaultAppendViewportHandler,
      pageSize: 20,
      placeholder: appendPlaceholder,
    })
  }, [mode])

  return (
    <div className="space-y-4">
      <div className="flex flex-wrap gap-3">
        <select
          className="h-9 rounded-md border bg-background px-3 text-sm"
          onChange={(event) => setMode(event.target.value as 'offset' | 'append')}
          value={mode}
        >
          <option value="offset">Offset mode</option>
          <option value="append">Append mode</option>
        </select>
        <select
          className="h-9 rounded-md border bg-background px-3 text-sm"
          onChange={(event) => {
            const nextCategory = event.target.value
            setCategory(nextCategory)
            model.send({ action: 'filter', payload: nextCategory })
          }}
          value={category}
        >
          <option value="all">All categories</option>
          <option value="Office">Office</option>
          <option value="Peripherals">Peripherals</option>
          <option value="Audio">Audio</option>
        </select>
      </div>

      <DataTable className="rounded-xl" model={model} style={{ height: 420 }}>
        <DataTableColumn field="name">
          <DataTableColumnHeader>Product</DataTableColumnHeader>
          <DataTableCell className="font-medium">{({ row }) => row.data.name}</DataTableCell>
        </DataTableColumn>
        <DataTableColumn field="category">
          <DataTableColumnHeader>Category</DataTableColumnHeader>
          <DataTableCell>{({ cellValue }) => String(cellValue)}</DataTableCell>
        </DataTableColumn>
        <DataTableColumn field="price">
          <DataTableColumnHeader className="justify-end">Price</DataTableColumnHeader>
          <DataTableCell className="text-right tabular-nums">{({ cellValue }) => currency.format(Number(cellValue ?? 0))}</DataTableCell>
        </DataTableColumn>
      </DataTable>
    </div>
  )
}
```

```ts live file=api.ts
import type { AppendFetchParams, FetchParams } from '@virtuoso.dev/data-table'

export interface Product {
  id: string
  name: string
  category: 'Office' | 'Peripherals' | 'Audio'
  price: number
}

export interface Query {
  category: string
}

const categories: Product['category'][] = ['Office', 'Peripherals', 'Audio']

const allProducts: Product[] = Array.from({ length: 240 }, (_, index) => ({
  id: `SKU-${String(index + 1).padStart(3, '0')}`,
  name: `${categories[index % categories.length]} Item ${index + 1}`,
  category: categories[index % categories.length]!,
  price: 49 + (index % 11) * 13,
}))

export const offsetPlaceholder: Product = {
  id: 'loading',
  name: 'Loading...',
  category: 'Office',
  price: 0,
}

export const appendPlaceholder = offsetPlaceholder

async function pause(ms: number, signal: AbortSignal) {
  await new Promise<void>((resolve, reject) => {
    const timer = window.setTimeout(resolve, ms)
    signal.addEventListener(
      'abort',
      () => {
        window.clearTimeout(timer)
        reject(new Error('aborted'))
      },
      { once: true }
    )
  })
}

function filterProducts(category: string) {
  return category === 'all' ? allProducts : allProducts.filter((product) => product.category === category)
}

export async function fetchOffsetProducts(params: FetchParams<Query>) {
  await pause(220, params.signal)
  const data = filterProducts(params.params.category)
  const rows = data.slice(params.offset, params.offset + params.limit)

  return {
    rows,
    totalCount: data.length,
  }
}

export async function fetchAppendProducts(params: AppendFetchParams<Query>) {
  await pause(220, params.signal)
  const data = filterProducts(params.params.category)
  const start = (params.cursor as number | undefined) ?? 0
  const rows = data.slice(start, start + params.limit)
  const nextCursor = start + rows.length

  return {
    cursor: nextCursor,
    hasMore: nextCursor < data.length,
    rows,
  }
}
```

## Notes

- Always honor the `AbortSignal` so stale viewport requests do not win a race.
- Placeholder rows are especially helpful in offset mode because the viewport can reserve space immediately.
