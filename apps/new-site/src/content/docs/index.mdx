## Quick Example

```tsx live
import { Virtuoso } from 'react-virtuoso'

export default function App() {
  return (
    <Virtuoso
      style={{ height: '400px' }}
      totalCount={1000}
      itemContent={(index) => (
        <div style={{ padding: '1rem', borderBottom: '1px solid #eee' }}>
          Item {index}
        </div>
      )}
    />
  )
}
```
